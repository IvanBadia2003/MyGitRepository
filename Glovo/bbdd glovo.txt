CREATE TABLE restaurante(
id_restaurante int auto_increment,
nombre varchar(20) not null,
descripcion varchar(250) not null,
puntuacion int(1),
ventas int(4) not null,
imagen varchar(250),
CONSTRAINT PK_RESTAURANTE primary key (id_restaurante)
);

CREATE TABLE producto(
id_producto int auto_increment,
nombre varchar(20) not null,
precio decimal(3,2) not null,
CONSTRAINT PK_PRODUCTO primary key (id_producto)
);

CREATE TABLE genero(
id_genero int auto_increment,
nombre varchar(20) not null,
CONSTRAINT PK_PRODUCTO primary key (id_genero)
);

CREATE TABLE producto_restaurante (
id_productorestaurante int auto_increment,
id_producto int(3) NOT NULL,
id_restaurante int(3) NOT NULL,
CONSTRAINT PK_PRODUCTO primary key (id_productorestaurante),
CONSTRAINT FK_PRODUCTO_RESTAURANTE foreign key (id_producto) references producto(id_producto),
CONSTRAINT FK_RESTAURANTE_PRODUCTO foreign key (id_restaurante) references restaurante(id_restaurante)
);

CREATE TABLE genero_restaurante (
id_generorestaurante int auto_increment,
id_genero int(3) NOT NULL,
id_restaurante int(3) NOT NULL,
CONSTRAINT PK_PRODUCTO primary key (id_generorestaurante),
CONSTRAINT FK_GENERO_RESTAURANTE foreign key (id_genero) references genero(id_genero),
CONSTRAINT FK_RESTAURANTE_GENERO foreign key (id_restaurante) references restaurante(id_restaurante)
);
